typedef struct node {
  char *line;
  struct node *prev;
  struct node *next;
} Node;

typedef struct TextEditor {
  Node *head;
  Node *tail;
  Node *cursor;
  int cursorLine;
  int cursorPos;
  DynamicArrayStack undoStack;
  DynamicArrayStack redoStack;
} TextEditor;

typedef struct {
  Action *actions;
  int size;
  int capacity;
} DynamicArrayStack;

Key Algorithms & Implementation

Text lines stored in doubly linked list for efficient insertion/deletion and bidirectional traversal.

Dynamic Array Stack

Undo/redo implemented using dynamically resizing arrays for action tracking.

String Search

// Example: Cursor movement implementation
int movecursorup(TextEditor *editor) {
  if(editor == NULL || editor->cursor == NULL) return FAILURE;
  if(editor->cursor->prev == NULL) return FAILURE;

  editor->cursor = editor->cursor->prev;
  editor->cursorLine--;

  int len = strlen(editor->cursor->line);
  if(editor->cursorPos > len) editor->cursorPos = len;

  return SUCCESS;
}

// Example: Dynamic stack push operation
int pushAction(DynamicArrayStack *stack, Action action) {
  if(stack->size == stack->capacity) {
    // Resize if full (double capacity)
    Action *new = malloc(sizeof(Action) * (2 * stack->capacity));
    for(int i=0; i<stack->size; i++) new[i] = stack->actions[i];
    free(stack->actions);
    stack->actions = new;
    stack->capacity = 2 * stack->capacity;
  }
  stack->actions[stack->size] = action;
  stack->size++;
  return SUCCESS;
}

Technical Challenges & Solutions

Bidirectional Navigation

Challenge: Efficient cursor movement between lines. Solution: Used doubly linked list for O(1) prev/next access.

Undo/Redo Memory

Challenge: Tracking all operations without excessive memory. Solution: Dynamic array stacks that resize as needed.

Boundary Handling

Challenge: Cursor movement at document edges. Solution: Comprehensive boundary checks in all navigation functions.

State Synchronization

Challenge: Keeping cursor position consistent after edits. Solution: Careful updating of line/position counters.
