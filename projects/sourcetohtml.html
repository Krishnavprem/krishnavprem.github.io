typedef enum {

typedef struct {
  pevent_e type;
  int property;
  int length;
  char data[PEVENT_DATA_SIZE];
} pevent_t;

typedef enum {
  PSTATE_IDLE,
  PSTATE_PREPROCESSOR_DIRECTIVE,
  PSTATE_HEADER_FILE,
  PSTATE_RESERVE_KEYWORD,
  PSTATE_NUMERIC_CONSTANT,
  PSTATE_STRING,
  PSTATE_SINGLE_LINE_COMMENT,
  PSTATE_MULTI_LINE_COMMENT,
  PSTATE_ASCII_CHAR
} pstate_e;

Parser State Machine

PSTATE_IDLE

Initial state, detects special characters to transition to other states.

PSTATE_PREPROCESSOR_DIRECTIVE

Handles #include, #define, #ifdef directives.

PSTATE_HEADER_FILE

Processes <header.h> or "header.h" files.

PSTATE_RESERVE_KEYWORD

Identifies C keywords (int, if, return, etc.).

PSTATE_NUMERIC_CONSTANT

Parses numbers (42, 3.14, 0xFF, etc.).

PSTATE_STRING

Handles string literals with escape sequences.

PSTATE_SINGLE_LINE_COMMENT

Processes // comments until newline.

PSTATE_MULTI_LINE_COMMENT

Handles /* */ comments across multiple lines.

PSTATE_ASCII_CHAR

Parses character literals like 'A', '\n'.

1
Character Reading: Read source file character by character in PSTATE_IDLE
2
State Transition: On special character (#, /, ", ', etc.), transition to appropriate state
3
Event Generation: State handler processes characters and generates parser events
4
HTML Conversion: Events are converted to HTML with appropriate CSS classes
5
File Output: Generated HTML written to output file with proper formatting

// Example: State transition logic
pevent_t * pstate_idle_handler(FILE *fd, int ch) {
  switch(ch) {
    case '/': // Comment detection
      int next_char = fgetc(fd);
      if(next_char == '/') {
        state = PSTATE_SINGLE_LINE_COMMENT;
      } else if(next_char == '*') {
        state = PSTATE_MULTI_LINE_COMMENT;
      }
      break;
    case '#': // Preprocessor directive
      state = PSTATE_PREPROCESSOR_DIRECTIVE;
      break;
    // ... more cases
  }
  return NULL;
}
